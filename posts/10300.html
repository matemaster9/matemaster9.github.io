select结构

基本结构

SELECT [field...]
FROM   [table...|view...]
WHERE  [condition]

常用的select结构

SELECT DISTINCT <*|select_list>
FROM <left_table> <join_type> JOIN <right_table>
ON <join_condition>
WHERE <where_condition>
GROUP BY <group_by_list> WITH<CUBE|ROLLUP> 
HAVING <having_condition>
ORDER BY <order_by_list> [ASC|DESC]
LIMIT [offset] <limit_number>

官方完整版结构

执行顺序

这里就不按照官方完整版结构去描述语句执行顺序，太多！太烦！太难！ 我们就只要熟练掌握常用的语句即可！在整个查询操作的过程中，每一次操作都会产生一个虚拟表（virtual table），只有最好的结果表会被返回。

（1）FROM：对关系表进行笛卡尔积，产生虚表vt1
（2）ON：对vt1做筛选，将满足条件的元组加入新的虚表vt2
（3）JOIN：对于指定了OUTER JOIN操作，那么完成对应的外连接操作，产生虚表vt3
（4）WHERE：对vt3元组进行条件筛选，将满足条件的元组加入新的虚表vt4
（5）GROUP BY：对vt4结果分组，如果存在 CUBE/ROLLUP操作，那么将操作处理后的结果加入vt5
（6）HAVING：根据过滤条件去除元组，产生虚表vt6
（7）SELECT：投影指定的列，如果存在DISTINCT操作，那么去除重复列后产生虚表vt7
（8）ORDER BY：根据排序的列进行升降序，产生虚表vt8
（9）LIMIT：根据偏移量/行数要求，产生vt9，并将结果表（result table）返回

分组查询

GROUP BY查询：根据s_id对fruits表中的数据进行分组

SELECT s_id, COUNT(*) AS Total
FROM fruits
GROUP BY s_id;
+------+-------+
| s_id | Total |
+------+-------+
|  101 |     3 |
|  103 |     2 |
|  104 |     2 |
|  107 |     2 |
|  102 |     3 |
|  105 |     3 |
|  106 |     1 |
+------+-------+

“注” ：这里通过一个聚集函数COUNT计算出每个分组含有的数目

当然我们还可以利用一个特殊的函数GROUP_CONCAT将每个分组的水果名称显示出来，并且通过HAVING子句对分组进行过滤，排除分组数目为一个的组的水果名称！

根据s_id对fruits表中的数据进行分组，并显示水果种类大于1的分组信息

mysql> SELECT s_id, GROUP_CONCAT(f_name) AS Names
    -> FROM fruits
    -> GROUP BY s_id
    -> HAVING COUNT(f_name) > 1;
+------+-------------------------+
| s_id | Names                   |
+------+-------------------------+
|  101 | apple,blackberry,cherry |
|  102 | orange,banana,grape     |

|  104 | berry,lemon             |
|  105 | melon,xbabay,xxtt       |
|  107 | xxxx,xbababa            |
+------+-------------------------+

左外连接（LEFT [OUTER] JOIN…ON…）右外连接（RIGHT [OUTER] JOIN…ON…）

全外连接：（LEFT [OUTER] JOIN…ON…） UNION （RIGHT [OUTER] JOIN…ON…）

子查询

子查询：利用嵌套的关系，使用内层select的结果去作为外层查询的条件实现查询功能！

一般情况，子查询会利用IN、ALL、ANY、SOME、EXISTS、θ一些关键字和比较运算符θ

先创建一个栗子

CREATE table tbl1 ( num1 INT NOT NULL);
CREATE table tbl2 ( num2 INT NOT NULL);
INSERT INTO tbl1 values(1), (5), (13), (27);
INSERT INTO tbl2 values(6), (14), (11), (20);
  • ANY/SOME

从tbl1中找出大于tbl2任意一个值的num1列元组

SELECT num1 
FROM tbl1 
WHERE num1 > ANY (SELECT num2 FROM tbl2);

+------+
| num1 |
+------+
|   13 |
|   27 |
+------+
  • ALL

从tbl1中找出大于tbl2所有值的num1列元组

SELECT num1 
FROM tbl1 
WHERE num1 > ALL (SELECT num2 FROM tbl2);

+------+
| num1 |
+------+
|   27 |
+------+
  • EXISTS

从供应商表中查找是否存在 107号供应商，若存在的话，找出由它供应的水果所有信息

select *
from fruits
where s_id = 107 and exists (select s_id from suppliers where s_id = 107);

+------+------+---------+---------+
| f_id | s_id | f_name  | f_price |
+------+------+---------+---------+
| b5   |  107 | xxxx    |    3.60 |
| t4   |  107 | xbababa |    3.60 |
+------+------+---------+---------+

我们通过子查询的结果判断：若查询的结果不为空，也就是EXISTS存在，那么就返回true，继续下一步外层查询

  • IN

在orderitems表中查询f_id为c0的订单号，并根据订单号查询具有订单号的客户c_id

SELECT c_id 
FROM orders 
WHERE o_num IN (SELECT o_num  FROM orderitems WHERE f_id = 'c0');

以上的步骤可以分为两步进行：

（1） 在订单信息中查询f_id为c0的订单号码

SELECT o_num  
FROM orderitems 
WHERE f_id = 'c0';

+-------+
| o_num |
+-------+
| 30003 |
| 30005 |
+-------+

（2）外层查询根据查询结果（30003,30005）进行下一步操作

SELECT c_id 
FROM orders 
WHERE o_num IN (30003, 30005);

# final result
+-------+
| c_id  |
+-------+
| 10004 |
| 10001 |
+-------+

查询由天津供应商供应的水果的所有信息

SELECT * 
FROM fruits
WHERE s_id = (SELECT s1.s_id FROM suppliers AS s1 WHERE s1.s_city = 'Tianjin');

WHERE vend_id = 1002 OR vend_id = 1003;

# AND 和 OR 的执行优先级（and优先）
## 因为优先级问题 下面两个select语句执行结果不一致！正确语义是下面
SELECT prod_name, prod_price
FROM products
WHERE vend_id = 1002 OR vend_id = 1003 AND prod_price <= 10;

SELECT prod_name, prod_price
FROM products
WHERE (vend_id = 1002 OR vend_id = 1003) AND prod_price <= 10;

# IN(values) in通过匹配括号中的值 返回true|false。下面两句等价
SELECT prod_name, prod_price
FROM products
WHERE vend_id IN (1002,1003);

SELECT prod_name, prod_price
FROM products
WHERE vend_id = 1002 OR vend_id = 1003);

# NOT 的全否定功能
SELECT prod_name, prod_price
FROM products
WHERE vend_id NOT IN (1002,1003);
  • LIKE匹配
# 找出所有以jet开头的产品(%任意长度的字符串)
SELECT prod_id, prod_name
FROM products
WHERE prod_name LIKE 'jet%';

# 找出以_ ton anvil（_任意一个字符）
SELECT prod_id, prod_name
FROM products
WHERE prod_name LIKE '_ ton anvil';
  • 正则表达式
# 二者都会匹配，但是LIKE不会返回结果，REGEXP返回结果
SELECT prod_id, prod_name
FROM products
WHERE prod_name LIKE '1000';

SELECT prod_id, prod_name
FROM products
WHERE prod_name REGEXP '1000';

# 如果需要实现正则表达式忽略大小写 BINARY
SELECT prod_id, prod_name
FROM products
WHERE prod_name REGEXP BINARY 'JetPack';

# 正则表达式中的OR
SELECT prod_id, prod_name
FROM products
WHERE prod_name REGEXP '1000|2000';

# 匹配几个字符之一(下面二者等价)
SELECT prod_id, prod_name
FROM products
WHERE prod_name REGEXP '[1,2,3] ton';

SELECT prod_id, prod_name
FROM products
WHERE prod_name REGEXP '[1|2|3] ton';  # 这里必须有[]，否则出错

SELECT prod_id, prod_name
FROM products
WHERE prod_name REGEXP '1|2|3 ton'; #这里匹配的是： 1 ，2 ，3 ton

# 范围查询（下面两者等价）
SELECT prod_id, prod_name
FROM products
WHERE prod_name REGEXP '[0123456789] ton';

SELECT prod_id, prod_name
FROM products
WHERE prod_name REGEXP '[0-9] ton';

# 匹配特殊字符 （可以使用\\转义特殊字符）
SELECT prod_id, prod_name
FROM products
WHERE prod_name REGEXP '.';

# 重复元字符（* ? + {n} {n,} {n,m}(m<=255)）
SELECT prod_name
FROM products
WHERE prod_name REGEXP '\\([0-9] sticks?\\)';

SELECT prod_name
FROM products
WHERE prod_name REGEXP '[[:digit:]]{4}';
  • 字段操作
SELECT concat(vend_name,'(',vend_country,')')
FROM vendors;

SELECT prod_id, quantity, item_price,
	   (quantity*item_price) AS expanded_price
FROM orderitems
WHERE order_num = 20005;
  • 函数处理
# 大写处理
SELECT vend_name, Upper(vend_name) AS vend_name_upcase
FROM vendors
ORDER BY vend_name;

# 时间处理
SELECT cust_id, order_num
FROM orders
WHERE Date(order_date) = '2005-09-01';

# 五个aggregate functions
SELECT AVG(prod_price) AS avg_price
FROM products;

SELECT COUNT(*) AS number_cust
FROM customers;

SELECT COUNT(cust_email) AS number_cust
FROM customers;

SELECT MAX(prod_price) AS max_price
FROM products;

SELECT MIN(prod_price) AS min_price
FROM products;

SELECT SUM(quantity) AS items_ordered
FROM orderitems
WHERE order_num = 20005;
  • 聚集函数
SELECT COUNT(*) AS items_quantity,
	   MAX(prod_price) AS price_max,
	   MIN(prod_price) AS price_min,
	   AVG(prod_price) AS price_avg,
	   SUM(prod_price) AS price_total
FROM products;
  • 相关子查询
SELECT cust_name
	   (
        SELECT COUNT(*)
        FROM orders
        WHERE orders.cust_id = customers.cust_id
       ) AS orders
FROM customers;
