插入排序

插入排序可视化

算法思想：默认第一个元素为已排序序列元素，选取未排序序列的元素，对已排序序列从后向前扫描，将当前为排序元素插入到已排序序列的合适位置

package cn.matemaster.algorithm.sorting;

public class Insertion {

    private static void exchange(Comparable[] arr, int i, int j) {
        Comparable temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }

    //-------降序版本------------
    private static boolean less(Comparable a, Comparable b) {
        return a.compareTo(b) < 0;
    }

    public static void desc(Comparable[] arr) {
        int n = arr.length;
        for (int i = 1; i < n; i++)
            for (int j = i; j >= 1; j--)
                if (less(arr[j-1], arr[j]))
                    exchange(arr, j-1, j);
    }

    //-------升序版本--------------
    private static boolean more(Comparable a, Comparable b) {
        return a.compareTo(b) > 0;
    }

    public static void asc(Comparable[] arr) {
        int n = arr.length;
        for (int i = 1; i < n; i++)
            for (int j = i; j >= 1; j--)
                if (more(arr[j-1], arr[j]))
                    exchange(arr, j-1, j);
    }

}

选择排序

选择排序可视化

算法思想：每次选择未排序序列中最值的元素，放入已排序序列的末尾，依此类推直到全部元素有序

package cn.matemaster.algorithm.sorting;

public class Selection {

    private static void exchange(Comparable[] arr, int i, int j) {
        Comparable temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }

    private static boolean less(Comparable a, Comparable b) {
        return a.compareTo(b) < 0;
    }

    private static boolean more(Comparable a, Comparable b) {
        return a.compareTo(b) > 0;
    }

    public static void desc(Comparable[] arr) {
        int n = arr.length;

        for (int i = 0; i < n; i++) {
            int max = i;
            for (int j = i+1; j < n; j++)
                if (more(arr[j], arr[max]))
                    max = j;
            exchange(arr, i, max);
        }

    }

    public static void asc(Comparable[] arr) {
        int n = arr.length;

        for (int i = 0; i < n; i++) {
            int min = i;
            for (int j = i+1; j < n; j++)
                if (less(arr[j], arr[min]))
                    min = j;
            exchange(arr, i, min);
        }
    }

}

冒泡排序

冒泡排序可视化

算法思想：它重复地走访过要排序的元素列，依次比较两个相邻的元素，如果顺序（如从大到小、首字母从Z到A）错误就把他们交换过来。走访元素的工作是重复地进行直到没有相邻元素需要交换，也就是说该元素列已经排序完成。

package cn.matemaster.algorithm.sorting;

import java.util.Arrays;

public class Bubble {
    
    private static void exchange(Comparable[] arr, int i, int j) {
        Comparable temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }

    private static boolean less(Comparable a, Comparable b) {
        return a.compareTo(b) < 0;
    }

    private static boolean more(Comparable a, Comparable b) {
        return a.compareTo(b) > 0;
    }

    public static void asc(Comparable[] arr) {
        int n = arr.length;

        for (int i = 0; i < n-1; i++) {
            for (int j = 0; j < n-1-i; j++) {
                if (more(arr[j], arr[j+1]))
                    exchange(arr, j, j+1);
            }
        }

    }
    
    public static void desc(Comparable[] arr) {
        int n = arr.length;

        for (int i = 0; i < n-1; i++) {
            for (int j = 0; j < n-1-i; j++) {
                if (less(arr[j], arr[j+1]))
                    exchange(arr, j, j+1);
            }
        }

    }
}

希尔排序

希尔排序可视化

算法思想：作为插入排序的优化算法，每次在增量gap间隔上保持有序，直到gap=1时，所有元素有序。

这里的SortTemplate类只是将上面的三个方法（less、more、exchange）封装在了一起！

package cn.matemaster.algorithm.sorting;

public class Shell {

    public static void asc(Comparable[] arr) {
        int n = arr.length;
        int gap = 1;
        // 确定增量:1、4、13、40、......
        while (gap < n/3)
            gap = gap*3 + 1;

        while (gap >= 1) {
            for (int i = gap; i < n ; i++) {
                for (int j = i; j >=gap ; j-=gap) {
                    if (SortTemplate.more(arr[j-gap], arr[j]))
                        SortTemplate.exchange(arr, j-gap, j);
                }
            }
            gap = gap/3;
        }
        
    }

    public static void desc(Comparable[] arr) {
        int n = arr.length;
        int gap = 1;
        while (gap < n/3)
            gap = gap*3 + 1;

        while (gap >= 1) {
            for (int i = gap; i < n ; i++) {
                for (int j = i; j >=gap ; j-=gap) {
                    if (SortTemplate.less(arr[j-gap], arr[j]))
                        SortTemplate.exchange(arr, j-gap, j);
                }
            }
            gap = gap/3;
        }
        
    }
}

归并排序

算法思想：归并操作可以将两个有序的左右子数组归并成一个有序的合数组。那么采取分治的思想。将排序序列分为左右子序列、归并，在对子序列进行同样的递归操作。

原地归并

package cn.matemaster.algorithm.sorting;

import java.util.Arrays;

public class Merge {

    private static Comparable[] aux;

    public static void merge(Comparable[] a, int lo, int mid, int hi) {

        // copy to aux[]
        System.arraycopy(a,0,aux,0,a.length);

        // merge back to a[]
        int i = lo, j = mid+1;
        for (int k = lo; k <= hi; k++) {
            if      (i > mid)                           a[k] = aux[j++];
            else if (j > hi)                            a[k] = aux[i++];
            else if (SortTemplate.more(aux[i], aux[j])) a[k] = aux[j++];
            else                                        a[k] = aux[i++];
        }
    }

    public static void setAux(int len) {
        aux = new Comparable[len];
    }

}

自顶向下归并排序

package cn.matemaster.algorithm.sorting;

import java.util.Arrays;

class MergeTD {
    private static void sort(Comparable[] a, int lo, int hi) {
        if (hi <= lo) return;
        int mid = lo + (hi - lo) / 2;
        sort(a, lo, mid);
        sort(a, mid + 1, hi);
        Merge.merge(a, lo, mid, hi);
    }

    public static void asc(Comparable[] arr) {
        Merge.setAux(arr.length);
        sort(arr,0, arr.length-1);
    }

}

自底向上归并排序

class MergeBU {

    public static void asc(Comparable[] arr) 
    {
        int n = arr.length;
        Merge.setAux(n);

        for (int sub = 1; sub < n; sub*=2) {
            for (int lo = 0; lo < n - sub; lo += sub+sub) {
                int mid = lo + sub -1;
                int hi = Math.min(lo + sub + sub -1, n-1);
                Merge.merge(arr,lo,mid,hi);
            }
        }       
    }
}

快速排序

算法思想：选取第一个元素为基准，将序列分为两个部分。重复切分操作，直到切分子序列为一个元素。

package cn.matemaster.algorithm.sorting;

import java.util.Arrays;

public class Quick {


    public static void asc(Comparable[] arr) {
        sort(arr, 0, arr.length-1);
    }

    // 先整个切分，再递归至左右子数组执行同样操作
    private static void sort(Comparable[] arr, int lo, int hi) {
        if (lo >= hi) return;
        int part = partition(arr,lo,hi);
        sort(arr, lo, part-1);
        sort(arr, part+1, hi);
    }


    private static int partition(Comparable[] arr, int lo, int hi) {
        // 扫描指针、切分值
        int i = lo;
        int j = hi+1;
        Comparable s = arr[lo];

        while (true) {
			// i 停止时指向的元素一定不大于s
            while (SortTemplate.more(s, arr[++i])) {
                if (i >= hi) break;
            }

            // j 停止时指向的元素一定不小于s
            while (SortTemplate.more(arr[--j], s)) {
                if (j <= lo) break;
            }

            if (i >= j) break;

            SortTemplate.exchange(arr, i, j);

        }
        // j 指向的元素就是需要切分的位置
        SortTemplate.exchange(arr, lo, j);
        return j;
    }

}

计数排序

针对有限元素集合S，以及线性表L

L中在S的元素si统计L中小于等于si的i个元素，那么元素就对应放在结果的线性表Ls的i位。

计数排序的基本思想：对于一个输入的元素m，确定小于m的元素个数。利用这一信息，我们可以将m放在输出数组的合适位置上。比如：小于m的元素个数为17个，那么我们就可以把m放在输出数组的第十八位。对于多个相同的m我们可以依次向后输出！

package cn.matemaster.algorithm.sorting;

import java.util.Arrays;

public class Count {
    public static void asc(int[] arr) {
        int max = arr[0], min = arr[0];
        int[] res = new int[arr.length];

        for (int i : arr) {
            if (i > max)
                max = i;
            if (i < min)
                min = i;
        }

        // 优化计数数组
        int[] count = new int[max - min + 1];

        // 对 arr元素计数
        for (int j = 0; j < arr.length; j++)
            count[arr[j] - min]++;

        // 确定归还位置(比m小的元素个数)
        for (int k = 1; k < count.length; k++)
            count[k] += count[k-1];

        // 排序到结果数组
        for (int l = arr.length; l > 0; l--)
            res[--count[arr[l-1] - min]] = arr[l-1];

        System.arraycopy(res, 0, arr, 0, arr.length);
        
    }

}

堆排序

堆排序（英语：Heapsort）是指利用堆这种数据结构所设计的一种排序算法。堆是一个近似完全二叉树的结构，并同时满足堆的性质：即子结点的键值或索引总是小于（或者大于）它的父节点。

因为堆的根结点一定是最值点，那么我们不断地进行取根、调整最大堆操作，就可以得到一个有序序列。这里的堆结构为了满足 floor(parent/2) = child，数组会从1号开始存储

堆有序的状态下：父结点一定大于子结点。若是状态被打破，那么说明一定是因为子结点变得比父结点大

package cn.matemaster.algorithm.sorting;

import java.util.Arrays;

public class Heap {

    private static void maxHeapify(Comparable[] arr, int start, int end) {
        // 父结点：parent、左结点：child
        int parent = start;
        int child = parent*2;

        while (child <= end) {
            // 选取大的子结点
            if (child + 1 <= end && SortTemplate.more(arr[child+1], arr[child]))
                child++;

            // 判断是否符合堆有序
            if (SortTemplate.more(arr[parent], arr[child]))
                return;
            else {
                SortTemplate.exchange(arr, parent, child);
                parent = child;
                child = parent*2;
            }
        }

    }

    private static void sort(Comparable[] arr) {
        int len = arr.length;

        // 自底向上调整堆
        for (int i = (len - 1)/2; i >= 1; i--)
            maxHeapify(arr, i, len -1);

        //  取最值、调整堆有序
        for (int j = len-1; j >= 1; j--) {
            SortTemplate.exchange(arr, 1, j);
            maxHeapify(arr, 1, j-1);
        }
    }

    public static void asc(Comparable[] arr) {
        // 辅助的堆 
        Comparable[] heap = new Comparable[arr.length+1];
        System.arraycopy(arr,0,heap, 1, arr.length);
        sort(heap);
        System.arraycopy(heap,1,arr, 0, arr.length);
    }

}
