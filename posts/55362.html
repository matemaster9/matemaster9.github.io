语法

建一个不带参数的存储过程

# 创建一个productpricing存储过程
CREATE PROCEDURE productpricing()
BEGIN
	SELECT Avg(prod_price) AS priceaverage
	FROM products;
END;

# 删除此存储过程
DROP PROCEDURE  IF EXISTS productpricing;

# 使用此存储过程（此时会输出结果@priceaverage）
CALL productpricing();
+--------------+
|@priceaverage |
+--------------+
|    16.133571 |
+--------------+

建一个带参数（输出）的存储过程

# 创建一个带参productpricing存储过程
CREATE PROCEDURE productpricing(
    OUT pMin DECIMAL(8,2),
    OUT pMax DECIMAL(8,2),
    OUT pAvg DECIMAL(8,2)
)
BEGIN
	SELECT Min(prod_price)
	FROM products
	INTO pMin;
	
	SELECT Max(prod_price)
	FROM products
	INTO pMax;
	
	SELECT Avg(prod_price)
	FROM products
	INTO pAvg;
END;

# 使用此存储过程（此时不在直接输出结果，而是将结果保存在输出参数中）
CALL productpricing(@pMin,@pMax,@pAvg);
SELECT @pMin,@pMax,@pAvg;
+-------+-------+-------+
| @pMin | @pMax | @pAvg |
+-------+-------+-------+
|  2.50 | 55.00 | 16.13 |
+-------+-------+-------+

# 删除此存储过程
DROP PROCEDURE  IF EXISTS productpricing;

这里解释一下：为什么不讲上面三个select子句合并？也就是下面的编写方式。

因为：在存储过程中，记录集不属于任何数据类型，为了和上面的输出参数类型保持一致，所以就必须将每一项的类型保持一一对应！

BEGIN
	SELECT Min(prod_price),Max(prod_price),Avg(prod_price)
	FROM products
	INTO pMin,pMax,pAcg;
END;

建一个智能的存储过程

场景描述：现在需要对订单进行统计，但是需要对某些地方的增加税收。

那么需要完成以下几步：

（1）正常统计订单

（2）对需要增加营业税的订单，进行税收处理

（3）返回统计的结果

存储过程：ordertotal

参数：

  • onumber：订单号
  • taxable ：是否税收
  • ototal：统计结果 
CREATE PROCEDURE ordertotal(
    IN onumber INT,
    IN taxable BOOLEAN,
    OUT ototal DECIMAL(8,2)
)COMMENT 'Obtain ordertotal,optional taxable'
BEGIN
	# 定义局部变量：total、taxrate
	DECLARE total DECIMAL(8,2);
	DECLARE taxrate INT DEFAULT 6;
	
	# 订单合计
	SELECT Sum(item_price*quantity)
	FROM orderitems
	INTO total;
	
	# 是否交税
	IF taxable THEN
		SELECT total+(total/100*taxrate) INTO total;
	END IF;
    
    # 返回结果
    SELECT total INTO ototal;
END;

使用上述的存储过程

# 不加税
CALL ordertotal(20005,0,@total);
SELECT @total;
+---------+
| @total  |
+---------+
| 1368.34 |
+---------+

# 加税
CALL ordertotal(20005,1,@total);
SELECT @total;
+---------+
| @total  |
+---------+
| 1450.44 |
+---------+

结构详解

CREATE PROCEDURE 存储过程名(参数列表) [修饰词] 过程主体

# 参数列表
[IN|OUT|INOUT] parameter type

# 修饰词
	LANGUAGE SQL #表明当前系统语言为SQL
	[NOT] DETERMINISTIC #无NOT表明：同样输入得到相同结果。NOT 表明：同样输入结果可能不一致

	{CONTAINS SQL|NO SQL|READS SQL DATA|MODIFIES SQL DATA} #子程序使用SQL的限制
	# CONTAINS SQL:子程序包含SQL，但是不包含读写的
	# NO SQL:子程序不包含SQL语句
	# READS SQL DATA:包含读数据SQK
	# MODIFIES SQL DATA:包含写数据SQL

	SQL SECURITY{DEFINER|INVOKER} #执行SQL的权限
	# DEFINER：定义者执行
	# INVOKER：有权限的调用者执行

	COMMENT 'String' # 注释信息，可以描述存储过程

# 过程主体
BEGIN
	# 主体SQL语句
END

存储函数

语法

CREATE FUNCTION 函数名(参数列表) RETURNS type [修饰词] 函数主体

# 参数列表和修饰词与存储过程保持一致，不做赘述

# 函数主体
RETURN value # value 可以是一个查询记录

这里定义一个SupplierName函数

CREATE FUNCTION SupplierName()
RETURNS CHAR(5)
RETURN (
    SELECT s_name
    FROM suppliers
    WHERE (s_call = '48075')
);

# 调用函数
SELECT SupplierName();

存储程序进阶

变量使用

之前在ordertotal存储过程中，使用到了局部变量的定义

DECLARE total DECIMAL(8,2);
DECLARE taxrate INT DEFAULT 6;

从上面代码看出，MySQL变量定义的方式

# 无默认值
DECLARE var_name var_type;
# 有默认值
DECLARE var_name var_type DEFAULT default_value;

那么在定义了变量之后，我们怎么实现对变量的赋值呢？

# set操作
SET var_name = value;

# select...into...
SELECT value INTO var_name;

## 这里的value可以是特定值或者查询结果！

定义条件和处理程序

定义条件和处理程序的好处在于：当存储程序发生错误或者问题时，可以使用定义的条件判断错误的类型，并且选择合适的处理程序进行问题处理，提高了存储程序能力，避免程序异常！

DECLARE condition_name CONDITION FOR condition_type;

# condition_name:错误的类型名，可以自定义
# condition_type:错误类型，分为：字符型（字符长度为5）、数值型错误代码
SQLSTATE [VALUE] : sqlstate_value|mysql_error_code

这里就直接举个栗子✌ ：定义一个错误代码 ERROR 1970(43990)，名称为command_not allowed的条件

# （1）sqlstate_value
DECLARE command_not_allowed CONDITION FOR 1970;
# （2）mysql_error_code
DECLARE command_not_allowed CONDOTION FOR '43990';

游标的使用

简单介绍一下游标cursor的声明、打开、使用、关闭

# 声明游标
DECLARE cursor_name CURSOR FOR select_statement;

# 打开游标
OPEN cursor_name;

# 使用游标
FETCH cursor_name INTO var_name;

# 关闭游标
CLOSE cursor_name;

简单地举个栗子✌

# 声明名称为cursor_fruit的光标
CREATE cursor_fruit CURSOR FOR (
    SELECT f_name, f_price 
    FROM fruits ; 
);

# 打开游标
OPEN cursor_fruit;

# 使用游标（变量必须在游标声明之前就已经定义了）
FETCH  cursor_fruit INTO fruit_name, fruit_price ;

# 关闭
CLOSE cursor_fruit;

流程控制

  • 理解if语句和if()函数的区别
IF condition1 THEN expression1
	[ELSE IF condition2 THEN expression2]
	[ELSE else_statement]
END IF;

举个栗子✌

IF var IS NULL THEN (SELECT 'var is null')
	ELSE (SELECT 'var is not null')
END IF;
  • case语句的两种形式
# 1.case存在变量
CASE vars
	WHEN var1 THEN expr1
	[WHEN var2 THEN expr2]
	[ELSE else_statement]
END CASE;	

# 2.case不存在变量
CASE
	WHEN condition1 THEN expr1
	[WHEN condition2 THEN expr2]
	[ELSE else_statement]
END CASE;

举个栗子✌

# (1)
CASE var
  WHEN 1 THEN SELECT 'val is 1';
  WHEN 2 THEN SELECT 'val is 2';
  ELSE SELECT 'val is not 1 or 2';
END CASE;

# (2)
CASE
  WHEN val IS NULL THEN SELECT 'val is NULL';
  WHEN val < 0 THEN SELECT 'val is less than 0';
  WHEN val > 0 THEN SELECT 'val is greater than 0';
  ELSE SELECT 'val is 0';
END CASE;
  • Loop循环
[loop_label]: LOOP
	loop_statement
END LOOP [loop_label]

举个栗子✌

# 求1~100之和
DECLARE s INT DEFAULT 0;
DECLARE id INT DEFAULT 0;
SUM : loop
	SET id = id + 1, s = s + id;
	IF id >= 100 THEN LEAVE SUM;
	END IF;
END LOOP SUM;

这里还涉及到一个LEAVE语句，用于跳出循环和BEGIN……END。当然还有一个用于直接跳转到循环头ITERATE。

CREATE PROCEDURE doiterate()
BEGIN
DECLARE p1 INT DEFAULT 0;
my_loop: LOOP
  SET p1= p1 + 1;
  IF p1 < 10 THEN ITERATE my_loop;
  ELSEIF p1 > 20 THEN LEAVE my_loop;
  END IF;
  SELECT 'p1 is between 10 and 20';
END LOOP my_loop;
END
  • REPEAT语句
[repeat_label] : REPEAT
	statement_list
UNTIL expr
END REPEAT [repeat_label]

举个栗子✌

DECLARE id INT DEFAULT 0;
REPEAT
SET id = id + 1;
UNTIL  id >= 10
END REPEAT;
  • WHILE语句
[while_label] : WHILE expr_condition
	DO statement
END WHILE [while_label]

举个栗子✌

DECLARE i INT DEFAULT 0;
WHILE i < 10 
	DO SET i = i + 1;
END WHILE;

存储程序的CRUD

# 更改存储程序
ALTER {PROCEDURE|FUNCTION} sp_name MODIFIES [alter_statement]

# 查看存储函数状态
SHOW {PROCEDURE|FUNCTION} STATUS [LIKE 'pattern']
SHOW CREATE {PROCEDURE|FUNCTION} sp_name

# 删除存储程序
DROP PROCEDURE|FUNCTION sp_name
