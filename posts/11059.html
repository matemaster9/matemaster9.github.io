DDL

创建SCT数据库

create database SCT;

创建关系表

student{ Sid char(8), Sname char(10), Ssex char(2), Sage int, Did char(2), Sclass char(6) }

create table student(
    Sid char(8),
    Sname char(10),
    Ssex char(2),
    Sage int,
    Did char(2),
    Sclass char(6)
)

dept{ Did char(2), Dname char(10), Dean char(10) }

create table dept(
    Did char(2),
    Dname char(10),
    Dean char(10)
)

course{ Cid char(3), Cname char(12), Chours int, Credit float(1), Tid char(3) }

create table course(
    Cid char(3),
    Cname char(12),
    Chours int,
    Credit float(1),
    Tid char(3)
)

teacher{ Tid char(3), Tname char(10), Did char(2), Salary float(2) }

create table teacher(
    Tid char(3),
    Tname char(10),
    Did char(2),
    Salary float(2)
)

sc{ Sid char(8), Cid char(3), Score float(1) }

create table sc(
    Sid char(8),
    Cid char(3),
    Score float(1)
)

DML

关于MySQL-select语句的使用，但是不介绍
这里先简单浏览一下，完整版本的MySQL-select系列语句结构！

select (select_list)
[into new_tb_name]
from (tb_name|view_name)
[where condition]
[group by expression]
[having search_condition]
[order by expression [desc|asc]]

其他DML

（1）insert语句可以参考源码中的数据插入！

这里演示一下向表中加入一个select检索结果（加入成绩及格的同学）：insert -> select

# 先创建newsc关系表
insert into SCT.newsc(Sid, Cid, Score)
	select Sid, Cid, Score
	from SC
	where (Score >= 60)

（2）根据where子句的条件，删除符合要求的元素（去除成绩不及格的同学记录）

delete from sc
where (Score < 60)

（3）update更新指定属性列的值（提高计算机教师的工资（1.5倍））

update teacher
set Salary = Salary(1+0.5)
where Did in 
	select Did
	from dept
	Where (Dname = '计算机')
