    public int thirdMax(int[] nums) {
        long first =Long.MIN_VALUE, second=Long.MIN_VALUE, third=Long.MIN_VALUE;
        for(int i = 0;i < nums.length; i++) {
            if(nums[i] >= first)
                first = nums[i];
        }
        for(int j = 0; j< nums.length; j++) {
            if(nums[j] < first&&nums[j] >= second)
                second = nums[j];
        }
        for(int k = 0; k< nums.length; k++) {
            if(nums[k] < second&&nums[k] >= third)
                third = nums[k];

        if(third == Long.MIN_VALUE)
            return (int)first;
        return (int)third;    
    }

class Solution {
    public int findShortestSubArray(int[] nums) {
        int len = nums.length;
        Map<Integer,int[]> map = new HashMap<>();
        for(int i=0;i<len;i++) {
            if(map.containsKey(nums[i])) {
                map.get(nums[i])[0]++;
                map.get(nums[i])[2] = i;
            } else {
                map.put(nums[i],new int[]{1,i,i});
            }
        }
        int maxDegree = 0, minLen = 0;
        for(Map.Entry<Integer,int[]> mp : map.entrySet()) {
            int[] value = mp.getValue();
            if(maxDegree<value[0]) {
                maxDegree = value[0];
                minLen = value[2] - value[1] +1;
            } else if (value[0] == maxDegree) {
                if(minLen > value[2] - value[1] +1) {
                    minLen = value[2] - value[1] +1;
                }
            }
        }
        return minLen;
    }
}

    public List<Integer> findDuplicates(int[] nums) {
        List<Integer> repeat = new ArrayList<>();
        int n = nums.length;
        for(int i = 0; i< n; i++) {
            if(nums[i] < n)
                nums[nums[i]%n-1] += n;
            else if (nums[i]%n == 0)
                nums[n-1] += n;
            else
                nums[nums[i]%n-1] += n;

        }
        for (int i = 0; i < n; i++) {
            if (nums[i] > 2*n)
                repeat.add(i+1);
        }
